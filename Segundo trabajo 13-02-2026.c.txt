#include <stdio.h>

int main() {
        float numero1, numero2, numero3, resta;

        printf("Ingrese el primer numero: ");
        scanf("%f" ,&numero1);

        printf("Ingrese el segundo numero: ");
        scanf("%f" ,&numero2);

        printf("Ingrese el tercer numero: ");
        scanf("%f" ,&numero3);

        resta= numero1- numero2- numero3;

        printf("El resultado de la resta es: %.2f ", resta);

        return 0;

}
